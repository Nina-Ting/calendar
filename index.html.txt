<script>
let y = new Date().getFullYear();
let m = new Date().getMonth();

function key(y,m,d){
  return `${y}-${m+1}-${d}`;
}

function load(){
  return JSON.parse(localStorage.getItem("events") || "{}");
}

function save(d){
  localStorage.setItem("events", JSON.stringify(d));
}

let editing = null;

function render(){
  const cal = document.getElementById("cal");
  cal.innerHTML = "";

  document.getElementById("title").innerText = `${y} / ${m+1}`;

  let data = load();

  let first = new Date(y,m,1);
  let start = (first.getDay()+6)%7;
  let days = new Date(y,m+1,0).getDate();

  const week = ["一","二","三","四","五","六","日"];

  week.forEach(w=>{
    let d=document.createElement("div");
    d.className="day-name";
    d.innerText=w;
    cal.appendChild(d);
  });

  for(let i=0;i<start;i++){
    cal.appendChild(document.createElement("div"));
  }

  for(let i=1;i<=days;i++){
    let box=document.createElement("div");
    box.className="day";

    let k = key(y,m,i);

    let date=document.createElement("div");
    date.className="date";
    date.innerText=i;
    box.appendChild(date);

    // ➕ 新增
    box.addEventListener("click",(e)=>{
      if(e.target.classList.contains("event")) return;

      let text = prompt("新增行程：");
      if(!text) return;

      let data = load();
      if(!data[k]) data[k]=[];

      data[k].push({
        id: Date.now(),
        text
      });

      save(data);
      render();
    });

    // 📌 顯示事件
    (data[k]||[]).forEach(item=>{
      let e=document.createElement("div");
      e.className="event";
      e.innerText=item.text;

      // ✏ Notion inline edit
      e.addEventListener("click",(ev)=>{
        ev.stopPropagation();

        let input=document.createElement("input");
        input.value=item.text;
        input.style.width="95%";

        e.innerHTML="";
        e.appendChild(input);
        input.focus();

        editing = { input, item, k };

        input.addEventListener("keydown",(e2)=>{
          if(e2.key==="Enter") saveEdit();
          if(e2.key==="Escape") cancelEdit();
        });

        document.addEventListener("click", outside);
      });

      box.appendChild(e);
    });

    cal.appendChild(box);
  }
}

/* ======================
   ✏ 編輯核心
====================== */

function saveEdit(){
  if(!editing) return;

  let data = load();
  let list = data[editing.k];
  let obj = list.find(x=>x.id==editing.item.id);

  let val = editing.input.value.trim();

  if(val===""){
    data[editing.k] = list.filter(x=>x.id!=editing.item.id);
  }else{
    obj.text = val;
  }

  save(data);
  cleanup();
  render();
}

function cancelEdit(){
  cleanup();
  render();
}

function outside(e){
  if(!editing) return;
  if(!e.target.closest(".event")){
    saveEdit();
  }
}

function cleanup(){
  document.removeEventListener("click", outside);
  editing=null;
}

function prevMonth(){
  m--;
  if(m<0){ m=11; y--; }
  render();
}

function nextMonth(){
  m++;
  if(m>11){ m=0; y++; }
  render();
}

render();
</script>
