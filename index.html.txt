  <button onclick="nextMonth()">下個月 ➡</button>

<div class="calendar" id="cal"></div>

<script>
let y = new Date().getFullYear();
let m = new Date().getMonth();
let editing = null;

function key(y,m,d){ return `${y}-${m+1}-${d}`; }
function load(){ return JSON.parse(localStorage.getItem("lyu_events") || "{}"); }
function save(d){ localStorage.setItem("lyu_events", JSON.stringify(d)); }

function render(){
  const cal = document.getElementById("cal");
  cal.innerHTML = "";
  document.getElementById("title").innerText = `${y} 年 ${m+1} 月`;

  let data = load();
  let first = new Date(y, m, 1);
  let start = (first.getDay() + 6) % 7; 
  let days = new Date(y, m + 1, 0).getDate();

  const week = ["一","二","三","四","五","六","日"];
  week.forEach(w => {
    let d = document.createElement("div");
    d.className = "day-name";
    d.innerText = w;
    cal.appendChild(d);
  });

  for(let i=0; i<start; i++) cal.appendChild(document.createElement("div"));

  for(let i=1; i<=days; i++){
    let box = document.createElement("div");
    box.className = "day";
    let k = key(y,m,i);

    let date = document.createElement("div");
    date.className = "date";
    date.innerText = i;
    box.appendChild(date);

    (data[k]||[]).forEach(item => {
      let e = document.createElement("div");
      e.className = "event";
      e.innerText = item.text;
      e.onclick = (ev) => {
        ev.stopPropagation();
        startEdit(e, item, k);
      };
      box.appendChild(e);
    });

    box.onclick = () => {
      let text = prompt("新增行程：");
      if(!text) return;
      let d = load();
      if(!d[k]) d[k] = [];
      d[k].push({ id: Date.now(), text });
      save(d);
      render();
    };
    cal.appendChild(box);
  }
}

function startEdit(el, item, k){
  let input = document.createElement("input");
  input.type = "text";
  input.value = item.text;
  el.innerHTML = "";
  el.appendChild(input);
  input.focus();
  editing = { input, item, k };

  input.onblur = saveEdit; 
  input.onkeydown = (e) => { if(e.key === "Enter") saveEdit(); };
}

function saveEdit(){
  if(!editing) return;
  let data = load();
  let val = editing.input.value.trim();
  if(val === ""){
    data[editing.k] = data[editing.k].filter(x => x.id != editing.item.id);
  } else {
    let obj = data[editing.k].find(x => x.id == editing.item.id);
    if(obj) obj.text = val;
  }
  save(data);
  editing = null;
  render();
}

function prevMonth(){ m--; if(m<0){ m=11; y--; } render(); }
function nextMonth(){ m++; if(m>11){ m=0; y++; } render(); }

render();
</script>
